      Here is the list of measures we will be using for the next steps:
      Total_Transactions_that_Have_Rent = COUNTROWS(FILTER(Dubai_RET, NOT(ISBLANK(Dubai_RET[rent_value])))),
      Total_Rent_Value = SUM(Dubai_RET[rent_value]),
      Average_Rent_Price = [Total_Rent_Value]/[Total_Transactions_that_Have_Rent]
      Average_Rent_YoY% = 
      VAR CurrentYearRent = [Average_Rent_Price]
      VAR PreviousYearRent =
          CALCULATE(
              [Average_Rent_Price],
              DATEADD('Calendar'[Date], -1, YEAR)
          )
      RETURN
          DIVIDE(
              CurrentYearRent - PreviousYearRent,
              PreviousYearRent
          )
      RentYoY%Arrow = SWITCH(TRUE(), [Average_Rent_YoY%]>0 ,UNICHAR(9650), UNICHAR(9660)),
      TransYoYArrow = SWITCH(TRUE(), [Total_Transactions YoY%]>0 ,UNICHAR(9650), UNICHAR(9660)),
      RevYoY%Arrow = SWITCH(TRUE(), [Total_RevenueYoY%]>0 ,UNICHAR(9650), UNICHAR(9660))

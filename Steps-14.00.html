                                           Step 66. Transactions by year Chart  
                                           
 
Now that we have our KPI’s situated, its time to start our visuals. Here is a List of Measures we will be using for the next few steps:
Average_Rent_Price = [Total_Rent_Value]/[Total_Transactions_that_Have_Rent],
Average_Rent_YoY% = 
VAR CurrentYearRent = [Average_Rent_Price]
VAR PreviousYearRent =
    CALCULATE(
        [Average_Rent_Price],
        DATEADD('Calendar'[Date], -1, YEAR)
    )
RETURN
    DIVIDE(
        CurrentYearRent - PreviousYearRent,
        PreviousYearRent
    ),
AverageRevenuePerMonth = DIVIDE([Total_Revenue], [DistinctMonths]),
AverageTransactionsPerMonth = DIVIDE([Total_Transactions], [DistinctMonths]),
Total_Revenue = SUM(Dubai_RET[Revenue]),
Total_RevenueYoY% = 
VAR CurrentYearRevenue = [Total_Revenue]
VAR PreviousYearRevenue = CALCULATE([Total_Revenue] , DATEADD('Calendar'[Date], -1, YEAR))
RETURN
DIVIDE(CurrentYearRevenue - PreviousYearRevenue, PreviousYearRevenue),
Total_Transactions = Count(Dubai_RET[Trans_Number]),
Total_Transactions YoY% = 
VAR CurrentYearTotalTransactions = [Total_Transactions]
VAR PreviousYearTotalTransactions = CALCULATE([Total_Transactions] , DATEADD('Calendar'[Date], -1, YEAR))
RETURN
DIVIDE(CurrentYearTotalTransactions - PreviousYearTotalTransactions, PreviousYearTotalTransactions)
